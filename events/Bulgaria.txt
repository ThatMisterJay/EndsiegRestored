add_namespace = bulgaria

# Salonica Armistice
country_event = {
	id = bulgaria.1
	title = bulgaria.1.t
	desc = bulgaria.1.d
	picture = GFX_report_event_salonica_armistice
	
	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		tag = BUL
		AND = {
			OR = {
				has_global_flag = compiegne	
				has_global_flag = mudros
				GRE = {
					controls_state = 106
				}		
			}
		}
	}		

	mean_time_to_happen = { 
		days = 1
	}		

	option = {
		name = bulgaria.1.a	
		news_event = { hours = 6 id = news.6 }
		set_global_flag = salonicaarmistice
		GER = { remove_from_faction = BUL }
		ROM = {
			transfer_state = 77
		}
		AUH = {
			transfer_state = 106
		}
		GRE = {
			transfer_state = 743
			transfer_state = 184
		}
	}
}

# Ferdinand I Abdicates
country_event = {
	id = bulgaria.2
	title = bulgaria.2.t
	desc = bulgaria.2.d
	picture = GFX_report_event_ferdinandgone
	
	fire_only_once = yes
	
	trigger = {
		NOT = { date > 1932.12.1 }
		tag = BNN
		has_global_flag = salonicaarmistice
	}
	
	mean_time_to_happen = { days = 4 }
	
	immediate = {
		set_global_flag = ferdinandgone
	}
	option = {
		name = bulgaria.2.a
		add_national_unity = 0.05
		# Boris III takes over
		retire_country_leader = yes
		create_country_leader = {
			name = "Boris III"
			desc = "POLITICS_BORIS_III_DESC"
			picture = "Portrait_Bulgaria_Boris.dds"
			expire = "1943.5.16"
			ideology = paternal_autocrat
			traits = {
				#
			}
		}
		BNN = {
			add_political_power = 100
		}
	}
}
