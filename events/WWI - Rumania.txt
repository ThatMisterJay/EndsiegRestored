# Treaty of Bucharest
country_event = {
	id = romania.1
	title = romania.1.t
	desc = romania.1.d
	picture = GFX_report_event_treaty_of_bucharest
	
	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		tag = RMM
		date > 1918.5.7
		NOT = { has_global_flag = bucharestreaty }
		NOT = { has_global_flag = welcome_event_1910 }
		NOT = { has_global_flag = welcome_event_1914 }
	}		

	mean_time_to_happen = { 
		days = 1
	}

	option = {
		name = romania.1.a
		news_event = { hours = 6 id = news.5 }
		set_global_flag = bucharestreaty
		BNN = {
			transfer_state = 77
			transfer_state = 765
		}
		AUH = {
			transfer_state = 748
		}
		RO2 = {
			annex_country = { target = RMM transfer_troops = yes }
			change_tag_from = RMM
		}
	}
}
