# Vardar Offensive
country_event = {
	id = greece.1
	title = greece.1.t
	desc = greece.1.d
	picture = GFX_report_event_vardar
	
	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		tag = GRE
		date > 1918.9.1
		NOT = { has_global_flag = vardar }
		AND = {
			GRE = {
				controls_state = 184
			}
		}
	}

	}

		name = greece.1.a
		set_global_flag = vardar
		GRE = {
			add_timed_idea = { idea = vardar_offensive days = 120 }
		}
	}
}
