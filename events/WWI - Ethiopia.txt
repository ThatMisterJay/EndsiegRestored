# Death of Menelik II

	id = ww1_ethiopia.1
	title = ww1_ethiopia.1.t
	desc = ww1_ethiopia.1.d
	picture = GFX_report_event_menelik
	
	fire_only_once = yes
	
	trigger = {
		date > 1913.12.12 
		has_global_flag = welcome_event_1910
		NOT = { date > 1914.1.1 }
	}

		name = ww1_ethiopia.1.a
		kill_country_leader = yes

			expire = "1943.5.16"
			ideology = paternal_autocrat
			traits = {
				#
			}
		}
	}
}
